SHELL_BANNER = """\
GNU MediaGoblin shell!
----------------------
Available vars:
 - mgoblin_app: instantiated mediagoblin application
 - mg_globals: mediagoblin.globals
 - db: database instance
"""


[docs]def py_shell(**user_namespace):
    """
    Run a shell using normal python shell.
    """
    code.interact(
        banner=SHELL_BANNER,
        local=user_namespace)




[docs]def ipython_shell(**user_namespace):
    """
    Run a shell for the user using ipython.
    """
    try:
        from IPython import embed
    except:
        print "IPython not available... exiting!"
        return
    
    embed(
        banner1=SHELL_BANNER,
        user_ns=user_namespace)




[docs]def shell(args):
    """
    Setup a shell for the user
    either a normal Python shell
    or an IPython one
    """
    user_namespace = {
        'mg_globals': mg_globals,
        'mgoblin_app': commands_util.setup_app(args),
        'db': mg_globals.database}

    if args.ipython:
        ipython_shell(**user_namespace)
    else:
        py_shell(**user_namespace)

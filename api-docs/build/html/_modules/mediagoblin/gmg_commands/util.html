[docs]def prompt_if_not_set(variable, text, password=False):
    """
    Checks if the variable is None and prompt for a value if it is
    """
    if variable is None:
        if not password:
            variable=raw_input(text + u' ')
        else:
            variable=getpass.getpass(text + u' ')
    
    return variable
